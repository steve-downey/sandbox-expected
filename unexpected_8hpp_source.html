35 // [expected.un.general] para 2: ill-formed instantiations
36 static_assert(std::is_object_v<E>, "unexpected<E>: E must be an object type (not void, reference, or function)");
37 static_assert(!std::is_array_v<E>, "unexpected<E>: E must not be an array type");
38 static_assert(std::is_same_v<E, std::remove_cv_t<E>>, "unexpected<E>: E must not be cv-qualified");
39 static_assert(!detail::is_unexpected_specialization<E>::value,
40 "unexpected<E>: E must not be a specialization of unexpected");

42 public:

46 template <class Err = E>
47 requires(!std::is_same_v<std::remove_cvref_t<Err>, unexpected> &&
48 !std::is_same_v<std::remove_cvref_t<Err>, std::in_place_t> && std::is_constructible_v<E, Err>)
49 constexpr explicit unexpected(Err&& e) noexcept(std::is_nothrow_constructible_v<E, Err>)
50 : unex_(std::forward<Err>(e)) {}
